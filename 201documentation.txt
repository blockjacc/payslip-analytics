Employeee Payroll_information Table			
table name	employee_payroll_information		
			
Field	Type	Description	
employee_payroll_information_id	int(11)	Primary key ID for the employee payroll information record	
emp_id	int(11)	Employee ID identifier 	employee_payroll_information.emp_id = employee.emp_id join to employee
company_id	int(11)	Company identifier	
department_id	int(11)	Unique identifier referencing the department from the department settings table	employee_payroll_information.department_id = department.dept_id get the department settings
rank_id	int(11)	Unique identifier referencing the employee rank from the rank settings table	employee_payroll_information.rank_id = rank.rank_id get the rank settings
payroll_group_id	int(11)	Identifier for the payroll group assigned to the employee	employee_payroll_information.payroll_group_id = payroll_group.payroll_group_id get the employment settings
employment_type	int(11)	References the employment type defined in the employment type settings table	employee_payroll_information.employment_type = employment_type.emp_type_id
position	int(11)	References the position title from the position settings table	employee_payroll_information.position= position.position_id get the project settings
cost_center	int(11)	Identifier for the cost center used for accounting and tracking (from settings)	employee_payroll_information.cost_center = cost_center.cost_center_id  on cost_center settings
project_id	int(11)	Unique identifier referencing the associated project from the project table	employee_payroll_information.project_id = project.project_id get the project settings
location_and_offices_id	int(11)	References the employee's assigned location or office from the locations table	employee_payroll_information.location_and_offices_id = location.location_and_offices get location
date_hired	date	Date the employee was hired	
last_date	date	Last working date (if applicable)	
payment_method	"enum('Cash','Debit','Electronic Transfer','Cheque','Bank Information') default null"	Mode of salary payment
